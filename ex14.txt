#include <stdio.h>
 
void imprime_serie(int i, int j, int k) {
    if (i > j)
      return;
    printf("%d ", i);
    imprime_serie(i + k, j, k);
}
 
int main() {
    int i, j, k;
    printf("Digite i, j e k: ");
    scanf("%d %d %d", &i, &j, &k);
    imprime_serie(i, j, k);
    printf("\n");
    return 0;
}
