#include <stdio.h>
 
void exibe_array(int arr[], int tamanho, int n) {
    if (n == tamanho) return;
    printf("%d ", arr[n]);
    exibe_array(arr, tamanho, n + 1);
}
 
int main(void) { 
    int x, i;
    printf("Digite o tamanho do array: ");
    scanf("%d", &x);
    int arr[x];
    for (i = 0; i < x; i++) {
        printf("arr[%d]: ", i);
        scanf("%d", &arr[i]);
    }
    printf("Array: ");
    exibe_array(arr, x, 0);
    return 0;
}
