#include <stdio.h>
 
void exibe_array(int arr[], int tamanho, int idx) {
    if (idx == tamanho) return;
    printf("%d ", arr[idx]);
    exibe_array(arr, tamanho, idx + 1);
}
 
int main(void) {
    int n, i;
    printf("Digite o tamanho do array: ");
    scanf("%d", &n);
    int arr[n];
    for (i = 0; i < n; i++) {
        printf("arr[%d]: ", i);
        scanf("%d", &arr[i]);
    }
    printf("Array: ");
    exibe_array(arr, n, 0);
    printf("\n");
    return 0;
}
