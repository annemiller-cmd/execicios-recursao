#include <stdio.h>
 
long long tetranacci(int n) {
    if (n == 0) return 0;
    if (n == 1) return 0;
    if (n == 2) return 0;
    if (n == 3) return 1;
    return tetranacci(n-1) + tetranacci(n-2) + tetranacci(n-3) + tetranacci(n-4);
}
 
int main(void) {
    int n;
    printf("Digite N: ");
    scanf("%d", &n);
    printf("Tetranacci(%d) = %lld\n", n, tetranacci(n));
    return 0;
}
