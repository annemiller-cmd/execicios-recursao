#include <stdio.h>
 
int potencia(int k, int n) {
    if (k == 0)
      return 0;
    if (n == 0)
      return 1;
    return k * potencia(k, n - 1);
}
 
int main() {
    int k, n;
    printf("Digite k e n: ");
    scanf("%d %d", &k, &n);
    printf("%d^%d = %d\n", k, n, potencia(k, n));
    return 0;
}
