#include <stdio.h>
 
long long fatorial(int n) {
    if (n == 0 || n == 1) return 1;
    return n * fatorial(n - 1);
}
 
/* Fatorial quadruplo de N = (2n)! / n! */
long long fatorial_quadruplo(int n) {
    return fatorial(2 * n) / fatorial(n);
}
 
int main(void) {
    int n;
    printf("Digite N: ");
    scanf("%d", &n);
    printf("Fatorial quadruplo de %d = %lld\n", n, fatorial_quadruplo(n));
    return 0;
}
