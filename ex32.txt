#include <stdio.h>
 
void decimal_para_binario(int n) {
    if (n == 0) return;
    decimal_para_binario(n / 2);
    printf("%d", n % 2);
}
 
int main(void) {
    int n;
    printf("Digite N: ");
    scanf("%d", &n);
    printf("Binario de %d: ", n);
    if (n == 0) printf("0");
    else decimal_para_binario(n);
    printf("\n");
    return 0;
}
